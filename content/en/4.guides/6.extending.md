This guide shows you how to extend the Auth Starter for advanced use cases without modifying the library source code. The starter follows the **Open/Closed Principle**: ready to use out-of-the-box, but open for extension.

## Custom User Fields

The starter provides a flexible way to add custom user data without modifying the database schema.

### Using JSONB Metadata

The `Profile` entity includes a `metadata` field (JSONB in PostgreSQL) for storing additional user information:

```java
package com.yourapp.service;

import dev.ctlabs.starter.auth.domain.model.User;
import dev.ctlabs.starter.auth.domain.repository.UserRepository;
import org.springframework.stereotype.Service;

import java.util.HashMap;
import java.util.Map;
import java.util.UUID;

@Service
public class UserProfileService {

    private final UserRepository userRepository;

    public UserProfileService(UserRepository userRepository) {
        this.userRepository = userRepository;
    }

    public void addCustomFields(UUID userId, Map<String, Object> customData) {
        User user = userRepository.findById(userId)
            .orElseThrow(() -> new RuntimeException("User not found"));

        // Get existing metadata or create new
        Map<String, Object> metadata = user.getProfile().getMetadata();
        if (metadata == null) {
            metadata = new HashMap<>();
        }

        // Add custom fields
        metadata.put("birthdate", customData.get("birthdate"));
        metadata.put("phoneNumber", customData.get("phoneNumber"));
        metadata.put("address", customData.get("address"));
        metadata.put("preferences", Map.of(
            "theme", "dark",
            "language", "en",
            "notifications", true
        ));

        user.getProfile().setMetadata(metadata);
        userRepository.save(user);
    }

    public Map<String, Object> getCustomFields(UUID userId) {
        User user = userRepository.findById(userId)
            .orElseThrow(() -> new RuntimeException("User not found"));

        return user.getProfile().getMetadata();
    }
}
```

**Usage:**
```java
@PostMapping("/me/profile")
public void updateProfile(
        @RequestBody Map<String, Object> customData,
        @AuthenticationPrincipal UserDetails userDetails) {

    UUID userId = getUserId(userDetails);
    userProfileService.addCustomFields(userId, customData);
}
```

## Overriding Beans

Replace default components with your own implementations.

### Custom Password Encoder

Replace BCrypt with Argon2:

```java
package com.yourapp.config;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.Primary;
import org.springframework.security.crypto.argon2.Argon2PasswordEncoder;
import org.springframework.security.crypto.password.PasswordEncoder;

@Configuration
public class CustomPasswordEncoderConfig {

    @Bean
    @Primary // Override the starter's default BCryptPasswordEncoder
    public PasswordEncoder passwordEncoder() {
        return Argon2PasswordEncoder.defaultsForSpringSecurity_v5_8();
    }
}
```

### Custom JWT Token Generator

Add custom claims to JWT tokens by extending the starter's `JwtService`:

::warning
**Important:** You must either extend `JwtService` or implement ALL its methods. The starter's authentication filter requires `extractAllClaims()` to validate tokens.
::

**Recommended approach - Extend JwtService:**

```java
package com.yourapp.security;

import dev.ctlabs.starter.auth.autoconfigure.AuthProperties;
import dev.ctlabs.starter.auth.infrastructure.security.JwtService;
import org.springframework.context.annotation.Primary;
import org.springframework.security.core.userdetails.UserDetails;
import org.springframework.stereotype.Service;

import java.util.HashMap;
import java.util.Map;

@Service
@Primary
public class CustomJwtService extends JwtService {

    public CustomJwtService(AuthProperties authProperties) {
        super(authProperties);
    }

    @Override
    public String generateToken(Map<String, Object> extraClaims, UserDetails userDetails) {
        // Add your custom claims
        Map<String, Object> customClaims = new HashMap<>(extraClaims);

        // Example: Add custom business data
        customClaims.put("organizationId", getCurrentOrganizationId());
        customClaims.put("accountTier", getCurrentAccountTier());
        customClaims.put("featureFlags", getFeatureFlags());

        // Call parent method to maintain validation logic
        return super.generateToken(customClaims, userDetails);
    }

    private String getCurrentOrganizationId() {
        // Your logic to get organization from context
        return "org-123";
    }

    private String getCurrentAccountTier() {
        // Your logic to get account tier
        return "premium";
    }

    private Map<String, Boolean> getFeatureFlags() {
        // Your logic to get feature flags
        return Map.of("advancedReports", true, "apiAccess", true);
    }
}
```

**Alternative - Implement all methods:**

```java
@Service
@Primary
public class CustomJwtService {

    private final AuthProperties authProperties;

    public CustomJwtService(AuthProperties authProperties) {
        this.authProperties = authProperties;
    }

    public String generateToken(Map<String, Object> extraClaims, UserDetails userDetails) {
        Map<String, Object> claims = new HashMap<>(extraClaims);
        claims.put("email", userDetails.getUsername());
        claims.put("customField", "customValue");

        return Jwts.builder()
                .claims(claims)
                .subject(userDetails.getUsername())
                .issuedAt(new Date(System.currentTimeMillis()))
                .expiration(new Date(System.currentTimeMillis() +
                    authProperties.getJwt().getExpiration().toMillis()))
                .signWith(getSignInKey())
                .compact();
    }

    // ⚠️ REQUIRED: Must implement this or authentication will fail
    public Claims extractAllClaims(String token) {
        return Jwts.parser()
                .verifyWith(getSignInKey())
                .build()
                .parseSignedClaims(token)
                .getPayload();
    }

    private SecretKey getSignInKey() {
        byte[] keyBytes = Decoders.BASE64.decode(authProperties.getJwt().getSecretKey());
        return Keys.hmacShaKeyFor(keyBytes);
    }
}
```

::tip{title="Best Practice"}
Extending `JwtService` is recommended because it ensures all required methods are implemented and validation logic remains consistent.
::
